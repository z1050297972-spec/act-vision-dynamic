## 6. 每个模型的缺点（按版本）

### 6.1 Baseline 缺点

- 高斯噪声 0%，鲁棒性崩溃。
- 遮挡仅 25%，关键区域被遮挡时策略失效明显。
- JPEG 压缩仅 38.5%，对压缩伪影非常敏感。
- 视觉域整体明显弱于动力学域（42% vs 82%）。

### 6.2 FT-v1 缺点

- 遮挡仅 50%，虽然翻倍但绝对值仍低。
- 联合扰动下降到 70%，出现视觉与动力学的负向协同。
- severe 视觉场景表现仍偏弱（约 63.2%）。

### 6.3 FT-v2 缺点

- 遮挡仍是 50%，连续两轮微调未继续改善。
- 视觉严重度 `severe` 仍非满分（约 78.9%）。
- 每类视觉类型样本量小（如遮挡 8 次），统计方差较大，应增加 targeted rollout 验证稳定性。
- 训练过程元数据不完整（缺训练日志），部分训练细节只能基于参数与文件状态推断。

## 7. 微调思路、步骤与方法（复盘）

## 7.1 Baseline -> FT-v1：先解决“看不清”

思路：

- 保持 ACT 主体结构不变，先通过训练时视觉增强补足感知鲁棒性。
- 使用课程学习避免一次性上强扰动导致训练不稳定。
- 对已知弱点（噪声、遮挡）加 focus。

方法（来自已有配置与代码）：

- 开启 `--train_visual_aug`
- 使用 `--train_aug_profile vision_robust_v2`
- 使用 `--train_aug_curriculum 3stage`
- 开启 `--train_aug_noise_focus`、`--train_aug_occlusion_focus`

结果：

- 噪声 0% -> 70%，是最核心突破。
- JPEG 38.5% -> 84.6%，显著提升。
- 遮挡 25% -> 50%，但仍是主短板。

## 7.2 FT-v1 -> FT-v2：在已有能力上定向强化

思路：

- 从 FT-v1 最优模型继续训练（迁移式微调），而非从头训练。
- 进一步强化视觉 profile（`vision_robust_v3`）。
- 降低学习率做“精修”。

方法（文档和脚本一致）：

- 使用 `--resume_ckpt .../ft_robust_20260220_145121/policy_best.ckpt`
- 使用 `--train_aug_profile vision_robust_v3`
- 保持 `3stage` 和 focus flags
- 学习率从 `1e-5` 调低到 `5e-6`（文档记录）

结果：

- Vision 总体到 90%，Dynamics 到 100%。
- 但遮挡依旧 50%，说明单纯增强强度提升不足以继续突破遮挡瓶颈。

def _apply_focus_flags(cfg, noise_focus=False, occlusion_focus=False):
    focused = _copy_aug_config(cfg)
    if noise_focus:
        focused['gaussian_noise_prob'] = min(1.0, focused['gaussian_noise_prob'] + 0.2)
        focused['gaussian_noise_sigma_max'] = min(50.0, focused['gaussian_noise_sigma_max'] + 6.0)
        focused['min_transforms'] = max(1, int(focused.get('min_transforms', 0)))
    if occlusion_focus:
        focused['occlusion_prob'] = min(1.0, focused['occlusion_prob'] + 0.15)
        focused['occlusion_area_max'] = min(0.45, focused['occlusion_area_max'] + 0.08)
        focused['occlusion_fill_mode'] = 'random'
        focused['min_transforms'] = max(1, int(focused.get('min_transforms', 0)))
    return focused

def build_train_aug_schedule(base_aug_config, curriculum, stage_ratios, noise_focus, occlusion_focus):
    if curriculum == 'off':
        return [_copy_aug_config(base_aug_config)], [1.0]

    if curriculum == '2stage':
        stage_cfgs = [_mild_aug_config(base_aug_config), _hard_aug_config(base_aug_config)]
    elif curriculum == '3stage':
        stage_cfgs = [_mild_aug_config(base_aug_config), _copy_aug_config(base_aug_config), _hard_aug_config(base_aug_config)]
    else:
        raise ValueError(f'Unsupported train_aug_curriculum: {curriculum}')

    stage_cfgs[-1] = _apply_focus_flags(stage_cfgs[-1], noise_focus=noise_focus, occlusion_focus=occlusion_focus)
    return stage_cfgs, stage_ratios

    policy.cuda()
    optimizer = make_optimizer(policy_class, policy)
    active_aug_stage = None
    train_dataset = getattr(train_dataloader, 'dataset', None)

    for epoch in tqdm(range(num_epochs)):
        if train_aug_stage_configs and train_dataset is not None:
            progress = (epoch + 0.5) / max(1, num_epochs)
            cumulative = 0.0
            stage_idx = len(train_aug_stage_configs) - 1
            for idx, ratio in enumerate(train_aug_stage_ratios):

def _sample_visual_cfgs(self):
    if not (self.is_train and self.train_aug):
        return []
    if self.rng.random() > self.aug_config['global_prob']:
        return []

    transform_probs = [
        ('brightness_contrast', float(self.aug_config['brightness_contrast_prob'])),
        ('gaussian_noise', float(self.aug_config['gaussian_noise_prob'])),
        ('occlusion', float(self.aug_config['occlusion_prob'])),
        ('motion_blur', float(self.aug_config['motion_blur_prob'])),
        ('jpeg_compression', float(self.aug_config['jpeg_prob'])),
        ('gamma', float(self.aug_config['gamma_prob'])),
        ('color_shift', float(self.aug_config['color_shift_prob'])),
    ]
    visual_cfgs = []
    for transform_type, prob in transform_probs:
        if self.rng.random() < prob:
            visual_cfgs.append(self._sample_cfg_for_type(transform_type))
    if visual_cfgs:
        self.rng.shuffle(visual_cfgs)
    return visual_cfgs

def _augment_image(self, image):
    visual_cfgs = self._sample_visual_cfgs()
    if not visual_cfgs:
        return image
    perturbed = image.copy()
    for visual_cfg in visual_cfgs:
        perturbed = apply_visual_perturbation(perturbed, visual_cfg, self.rng)
    return perturbed

def build_robust_scenarios(mode, suite, seed, num_rollouts):
    rng = np.random.default_rng(seed)
    scenarios = []
    for rollout_idx in range(num_rollouts):
        scenario_seed = int(rng.integers(0, np.iinfo(np.int32).max))
        scenario_rng = np.random.default_rng(scenario_seed)
        scenario = {
            'scenario_id': f'{mode}_{rollout_idx:04d}',
            'mode': mode,
            'suite': suite,
            'seed': scenario_seed,
            'visual_cfg': None,
            'dynamics_cfg': None,
        }
        if mode == 'vision_only':
            scenario['visual_cfg'] = _build_visual_cfg(suite, scenario_rng)
        elif mode == 'dynamics_only':
            scenario['dynamics_cfg'] = _build_dynamics_cfg(suite, scenario_rng)
        elif mode == 'vision_dynamics':
            scenario['visual_cfg'] = _build_visual_cfg(suite, scenario_rng)
            dynamics_suite = _paired_dynamics_suite_for_vision_suite(suite)
            scenario['dynamics_cfg'] = _build_dynamics_cfg(dynamics_suite, scenario_rng)
        scenarios.append(scenario)
    return scenarios

def apply_visual_perturbation(image, cfg, rng):
    perturbation_type = cfg.get('type')
    if perturbation_type == 'gaussian_noise':
        sigma = float(cfg['sigma'])
        noise = rng.normal(loc=0.0, scale=sigma, size=image.shape)
        image_f = image.astype(np.float32) + noise
        return np.clip(image_f, 0.0, 255.0).astype(np.uint8)

    if perturbation_type == 'motion_blur':
        kernel_size = int(cfg['kernel_size'])
        return _apply_motion_blur(image, kernel_size)

    if perturbation_type == 'brightness_contrast':
        alpha = float(cfg['alpha'])
        beta = float(cfg['beta'])
        image_f = image.astype(np.float32) * alpha + beta
        return np.clip(image_f, 0.0, 255.0).astype(np.uint8)

    if perturbation_type == 'occlusion':
        h, w, _ = image.shape
        area_ratio = float(cfg['area_ratio'])
        fill_mode = str(cfg.get('fill_mode', '')).lower()
        target_area = max(1, int(round(h * w * area_ratio)))
        ...
        if fill_mode == 'random':
            random_block = rng.integers(0, 256, size=(occ_h, occ_w, 3), dtype=np.uint8)
            perturbed[top:top + occ_h, left:left + occ_w, :] = random_block
        elif fill_mode == 'mean':
            mean_color = perturbed.reshape(-1, 3).mean(axis=0)
            perturbed[top:top + occ_h, left:left + occ_w, :] = mean_color.astype(np.uint8)
        else:
            perturbed[top:top + occ_h, left:left + occ_w, :] = fill_value
        return perturbed

    if perturbation_type == 'jpeg_compression':
        quality = int(cfg['quality'])
        return _apply_jpeg_compression(image, quality)
```

def apply_dynamics_randomization(env, task_name, dynamics_cfg):
    if dynamics_cfg is None:
        return None

    if 'sim_insertion' in task_name:
        target_geoms = INSERTION_DYNAMICS_GEOMS
    elif 'sim_transfer_cube' in task_name:
        target_geoms = TRANSFER_DYNAMICS_GEOMS
    else:
        return None

    model = env.physics.model
    body_mass_scales = dynamics_cfg.get('body_mass_scales', {})
    for body_name, scale in body_mass_scales.items():
        body_id = _safe_name2id(model, body_name, 'body')
        if body_id is None:
            continue
        new_mass = max(1e-4, float(model.body_mass[body_id]) * float(scale))
        model.body_mass[body_id] = new_mass

    friction_scale = float(dynamics_cfg.get('friction_scale', 1.0))
    for geom_name in target_geoms:
        geom_id = _safe_name2id(model, geom_name, 'geom')
        if geom_id is None:
            continue
        new_slide = max(1e-4, float(model.geom_friction[geom_id, 0]) * friction_scale)
        model.geom_friction[geom_id, 0] = new_slide

    dof_damping_scale = float(dynamics_cfg.get('dof_damping_scale', 1.0))
    model.dof_damping[:] = np.maximum(model.dof_damping[:] * dof_damping_scale, 1e-5)

    actuator_gain_scale = float(dynamics_cfg.get('actuator_gain_scale', 1.0))
    model.actuator_gainprm[:, 0] *= actuator_gain_scale
    model.actuator_biasprm[:, 1] *= actuator_gain_scale
```
